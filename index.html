// Helper function to get URL parameters function getUrlParam(param) { const params = new URLSearchParams(window.location.search); return params.get(param); } // Load cart from localStorage or empty array let cart = JSON.parse(localStorage.getItem('store_cart') || '[]'); // Function to save cart to localStorage function saveCart() { localStorage.setItem('store_cart', JSON.stringify(cart)); } // Add new product from URL parameters if available function addProductFromUrl() { const name = getUrlParam('add'); const price = parseFloat(getUrlParam('price')); if (name && !isNaN(price)) { // Check if product already in cart const existing = cart.find(item => item.name === name); if (existing) { existing.qty += 1; } else { cart.push({ name: name, price: price, qty: 1 }); } saveCart(); // Clear URL parameters so it doesn't add again on refresh history.replaceState(null, '', window.location.pathname); } } // Render cart HTML function renderCart() { const container = document.getElementById('cartItemsContainer'); container.innerHTML = ''; if (cart.length === 0) { container.innerHTML = '

${item.name}
Price: Rs. ${item.price.toFixed(2)}

`; container.appendChild(div); }); updateTotals(subtotal); } function updateTotals(subtotal) { const deliveryCharge = subtotal > 0 ? 200 : 0; document.getElementById('subtotalText').textContent = `Rs. ${subtotal.toFixed(2)}`; document.getElementById('deliveryText').textContent = `Rs. ${deliveryCharge.toFixed(2)}`; document.getElementById('totalText').textContent = `Rs. ${(subtotal + deliveryCharge).toFixed(2)}`; } function changeQty(index, delta) { cart[index].qty += delta; if (cart[index].qty <= 0) { cart.splice(index, 1); } saveCart(); renderCart(); } function removeItem(index) { cart.splice(index, 1); saveCart(); renderCart(); } function clearCart() { if (confirm('Clear cart?')) { cart = []; saveCart(); renderCart(); } } // On page load window.onload = function() { addProductFromUrl(); renderCart(); // Setup event listeners for clear cart button document.getElementById('clearCartBtn').addEventListener('click', clearCart); // Add your other button listeners like checkout here };
